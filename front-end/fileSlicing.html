    const blobArr = []

        if (contentDisposition) {

        }
      }
      blobArr.push({

    }

    blobArr.sort((a, b) => a.index - b.index)

    // å¤šä¸ªblobæ’åºå®Œåˆå¹¶ä¸ºä¸€ä¸ªblob
    const buffers = new Blob(targetBlobArr)

}

function getFileSize (param, sizeUrl) {
  return new Promise((resolve, reject) => {
    request({
      url: sizeUrl,
      method: 'get',
      params: param
    }).then((size) => {
      if (size) {
        resolve(size)
      } else {
        reject(size)
      }
    })
      .catch((e) => {
        reject(e)
      })
  })
}

function saveAs (name, buffers, callback) {
  const blobUrl = URL.createObjectURL(buffers)
  const a = document.createElement('a')
  a.download = name
  a.href = blobUrl
  a.click()
  callback && callback()
  // URL.revokeObjectURL(blob)
}
